    init pubnub

    const pubnub = new PubNub({
  publishKey: PUBLISH_KEY,
  subscribeKey: SUBSCRIBE_KEY,
  authKey: AUTHORIZATION_TOKEN,
  ssl: true
});

const message = {
  CONNECTED: 'dog'
};

const channel = MESSAGE_CHANNEL;

console.info(`sending message: ${JSON.stringify(message)}`);

pubnub.publish({
  channel,
  message
}, ({ error }) => {
  if (error) {
    console.error('error occured sending message!');
    process.exit(1);
  }

  console.info('sent message');
  process.exit(0);
});
