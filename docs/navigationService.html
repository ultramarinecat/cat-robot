    get average of the set of proximities

    /* eslint-disable no-shadow */
function averageProximity(proximities) {
  /* istanbul ignore if */
  if (proximities == null || proximities.size === 0) {
    return 0;
  }

  let sum = 0;
  for (const proximity of proximities) {
    sum += proximity;
  }

  return sum / proximities.size;
}
